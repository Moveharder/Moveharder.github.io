  startAudio(url, is_remote = false) {
    // !!! ä½¿ç”¨çš„æ—¶å€™å†åˆ›å»ºï¼Œå› ä¸ºåœ¨å¤šç«¯åº”ç”¨æ¨¡å¼ä¸­ï¼Œä¼šå‡ºç°è°ƒç”¨audioSource.start()ä¸ä¼šæ’­æ”¾çš„é—®é¢˜
    this.createWebAudioCtx(this.fftSize);

      .then((buffer) => {

  startAnimate() {
    let {
      ctx,
      canvas,
      canvasWidth,
      canvasHeight,
      analyser,
      barWidth,
      barHeightScale,
      barMargin,
      horizonPadding,
      normalizedBuffer,
    } = this;

    // è·å–éŸ³é¢‘æ•°æ®
    let bufferLength = analyser.frequencyBinCount;
    let dataArray = new Uint8Array(bufferLength);

    // åŠ¨ç”»å‡½æ•°
    const animate = () => {
      ctx.clearRect(
        -canvasWidth / 2,
        -canvasHeight / 2,
        canvasWidth,
        canvasHeight
      );

      // è·å–éŸ³é¢‘æ•°æ®
      analyser.getByteFrequencyData(dataArray);
      let normalizedArr = normalizedBuffer(dataArray);
      // normalizedArr = normalizedArr.filter(item => item > 0.3)

      const barCount = Math.ceil(canvasWidth / (barWidth + barMargin));
      const halfBarCount = Math.floor(barCount / 2);
      const barStep = Math.floor(bufferLength / barCount);

      // ç»˜åˆ¶éŸ³æ³¢æŸ±çŠ¶æ¡
      for (let i = -halfBarCount; i <= halfBarCount; i++) {
        let index = Math.abs(i) * barStep;
        let item = normalizedArr[index];
        let barHeight = Math.round(item * barHeightScale); // å ä½é«˜åº¦
        let x = i * (barMargin + barWidth) - (barMargin + barWidth) / 2;
        let y = Math.ceil(-barHeight / 2); //å‚ç›´å±…ä¸­

        // æ’é™¤å·¦å³è¾¹è·èŒƒå›´å†…çš„æ¸²æŸ“
        if (
          x > -canvasWidth / 2 + horizonPadding &&
          x < canvasWidth / 2 - horizonPadding
        ) {
          this.drawItem(x, y, barWidth, barHeight);
        }
      }

      // ç»§ç»­ä¸‹ä¸€å¸§åŠ¨ç”»
      this.animationId = canvas.requestAnimationFrame(animate);
    };

    // å¼€å§‹åŠ¨ç”»å¾ªç¯
    animate();
  }

    }

  drawItem(x, y, w, h, opacity = 1) {
    let baseFixedY = 0.5;
    let baseFixedW = 0.5;
    let radius = w / 2 - baseFixedW;
    opacity = Math.max(0.1, opacity);
    this.drawCircle(
      x,
      h / 2 - baseFixedY,
      radius,
      0,
      Math.PI,
      this.getBarColor(opacity)
    );
    this.drawRect(x, y, w, h, this.getBarColor(opacity));
    this.drawCircle(
      x,
      -h / 2 + baseFixedY,
      radius,
      Math.PI,
      2 * Math.PI,
      this.getBarColor(opacity)
    );
  }

  drawCircle(
    x,
    y,
    radius,
    startAngle = 0,
    endAngle = 2 * Math.PI,
    color = "#ffffff"
  ) {
    this.ctx.beginPath();
    this.ctx.strokeStyle = color;
    this.ctx.fillStyle = color;
    this.ctx.arc(x, y, radius, startAngle, endAngle, false);
    this.ctx.stroke();
    this.ctx.fill();
    this.ctx.closePath();
  }

  drawRect(x, y, w, h, color = "#ffffff") {
    this.ctx.strokeStyle = color; //this.getBarColor(opacity);
    this.ctx.fillStyle = color; //this.getBarColor(opacity);
    this.ctx.fillRect(x - w / 2, y, w, h);
  }

  /**
   * åŠ è½½éŸ³é¢‘æ–‡ä»¶bufferæ•°æ®
   * @param {String} url éŸ³é¢‘åœ°å€
   * @param {Boolean} is_remote æ˜¯å¦æ˜¯åœ¨çº¿åœ°å€
   */
  loadAudio(url, is_remote = false) {
    const { audioContext } = this;
    return new Promise((resolve, reject) => {
      if (is_remote) {
        // å¤„ç†åœ¨çº¿æ–‡ä»¶
        wx.request({
          url,
          responseType: "arraybuffer",
          success: (res) => {
            audioContext.decodeAudioData(
              res.data,
              (buffer) => {
                resolve(buffer);
              },
              (err) => {
                console.error("decodeAudioData fail", err);
                reject(err);
              }
            );
          },
          fail: (err) => {
            console.error("request fail", err);
            reject(err);
          },
        });
      } else {
        // å¤„ç†æœ¬åœ°æ–‡ä»¶
        this.fs.readFile({
          filePath: url,
          success: (res) => {
            // console.log('åŠ è½½éŸ³é¢‘æ•°æ®ï¼š', res.data)
            audioContext.decodeAudioData(
              res.data,
              (buffer) => {
                // console.log('éŸ³é¢‘æ•°æ®è§£ç ï¼š', buffer)
                resolve(buffer);
              },
              (err) => {
                console.error("decodeAudioData fail", err);
                reject(err);
              }
            );
          },
          fail: (err) => {
            console.error("err:", err);
            reject(err);
          },
        });
      }
    });
  }

  getBarColor(opacity = 1) {
    return `rgba(255, 100 ,230, ${opacity})`;
  }

  normalizedBuffer(data = []) {
    let copyData = [...data];
    // æ‰¾åˆ°éŸ³é¢‘æ•°æ®çš„æœ€å¤§å€¼å’Œæœ€å°å€¼
    const max = Math.max(...copyData);
    const min = Math.min(...copyData);

    // è®¡ç®—éŸ³é¢‘æ•°æ®çš„èŒƒå›´
    const range = max - min;
    // console.log(min, max, range)

    // å¯¹éŸ³é¢‘æ•°æ®è¿›è¡Œå½’ä¸€åŒ–å¤„ç†ï¼ŒéŸ³é¢‘æ•°æ®èŒƒå›´åœ¨ 0 åˆ° 1 ä¹‹é—´
    return copyData.map((sample) => (sample - min) / range || 0);
  }

  handleError(err) {
    this.onError && this.onError(err);
  }
